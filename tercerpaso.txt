Objetivo

Implementar un sistema de navegación autónoma que permita al coche seguir una línea en el suelo mediante el uso de un sensor infrarrojo.

⚙️ Concepto de funcionamiento

El módulo de seguimiento de línea utiliza un sensor IR para detectar diferencias de color basadas en la reflexión:

Superficie oscura (negra) → absorbe luz → detectada como línea
Superficie clara (blanca) → refleja luz → detectada como fondo

El Arduino lee constantemente este valor y ajusta el movimiento del coche en tiempo real.

Lá lógica detras del functionamiento es que el arduino hace que el motor siga una linea y tambien realiza pequeños giros(ajustes) cuando es necesario para volver a la linea.

##nueva material usado: Sensor de linea basado de Infrarojo

##código
const int a1 = 6; 
const int a2 = 5;
const int b1A = 10;
const int b2A = 9;

const int ir_left = 2;
const int ir_right = 13;

int speed = 110;
int turn_speed = 110;
int kick = 140;
int kick_duration = 20;

int last_position = 0;
int last_action = -1; // tracks what the car was doing last iteration


void moveforward() {
  analogWrite(a1, speed);
  analogWrite(b2A, speed);
  digitalWrite(a2, 0);
  digitalWrite(b1A, 0);
}
void turn_left() {
  analogWrite(a1, turn_speed);
  analogWrite(a2, 0);
  analogWrite(b1A, 0);
  analogWrite(b2A, 0);
}
void turn_right() {
  analogWrite(b1A, 0);
  analogWrite(b2A, turn_speed);
  analogWrite(a1, 0);
  analogWrite(a2, 0);
}
void stop_move() {
  analogWrite(a1, 0);
  analogWrite(a2, 0);
  analogWrite(b1A, 0);
  analogWrite(b2A, 0);
}
void kickstart(int action) {
  // only kick if the action has changed
  if (action != last_action) {
    if (action == 0) { // forward kick
      analogWrite(a1, kick);
      analogWrite(b2A, kick);
      digitalWrite(a2, 0);
      digitalWrite(b1A, 0);
    } else if (action == 1) { // left kick
      analogWrite(a1, kick);
      analogWrite(a2, 0);
      analogWrite(b1A, 0);
      analogWrite(b2A, 0);
    } else if (action == 2) { // right kick
      analogWrite(b2A, kick);
      analogWrite(b1A, 0);
      analogWrite(a1, 0);
      analogWrite(a2, 0);
    }
    delay(kick_duration);
    last_action = action;
  }
}

void setup() {
  Serial.begin(9600);
  pinMode(a1, OUTPUT);
  pinMode(a2, OUTPUT); 
  pinMode(b1A, OUTPUT);
  pinMode(b2A, OUTPUT);
  pinMode(ir_left, INPUT);
  pinMode(ir_right, INPUT);
}

void loop() {
  int left = digitalRead(ir_left);
  int right = digitalRead(ir_right);

  Serial.print("L: ");
  Serial.print(left);
  Serial.print(" | R: ");
  Serial.println(right);

  if (left == 1 && right == 1) {
    kickstart(0);
    moveforward();
    last_position = 0;
  }
  else if (left == 0 && right == 1) {
    kickstart(1);
    turn_left();
    last_position = -1;
  }
  else if (left == 1 && right == 0) {
    kickstart(2);
    turn_right();
    last_position = 1;
  }
  else if (left == 0 && right == 0) {
    if (last_position == -1) {
      kickstart(1);
      turn_left();
    } else if (last_position == 1) {
      kickstart(2);
      turn_right();
    } else {
      kickstart(3);
      stop_move();
      last_action = 3;
    }
  }
  delay(10);
}
