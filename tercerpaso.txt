Objetivo

Implementar un sistema de navegación autónoma que permita al coche seguir una línea en el suelo mediante el uso de un sensor infrarrojo.

⚙️ Concepto de funcionamiento

El módulo de seguimiento de línea utiliza un sensor IR para detectar diferencias de color basadas en la reflexión:

Superficie oscura (negra) → absorbe luz → detectada como línea
Superficie clara (blanca) → refleja luz → detectada como fondo

El Arduino lee constantemente este valor y ajusta el movimiento del coche en tiempo real.

Lá lógica detras del functionamiento es que el arduino hace que el motor siga una linea y tambien realiza pequeños giros(ajustes) cuando es necesario para volver a la linea.

##nueva material usado: Sensor de linea basado de Infrarojo

##código
// Motor pins
const int a1 = 6;
const int a2 = 5;
const int b1A = 10;
const int b2A = 9;

// IR sensors
const int ir_left = 13;
const int ir_right = 2;

// Speed settings
int speed = 150;
int turn_speed = 100;

// Last known direction
// -1 = left, 1 = right, 0 = straight
int last_position = 0;

void setup() {
  pinMode(a1, OUTPUT);
  pinMode(a2, OUTPUT);
  pinMode(b1A, OUTPUT);
  pinMode(b2A, OUTPUT);

  pinMode(ir_left, INPUT);
  pinMode(ir_right, INPUT);

  Serial.begin(9600); // for debugging
}

void loop() {
  int left = digitalRead(ir_left);
  int right = digitalRead(ir_right);

  Serial.print("L: "); Serial.print(left);
  Serial.print(" R: "); Serial.println(right);

  // BOTH ON LINE → go straight
  if (left == 0 && right == 0) {
    moveForward();
    last_position = 0;
  }

  // LEFT SENSOR OFF → drifted right → turn LEFT
  else if (left == 1 && right == 0) {
    turnLeft();
    last_position = -1;
  }

  // RIGHT SENSOR OFF → drifted left → turn RIGHT
  else if (left == 0 && right == 1) {
    turnRight();
    last_position = 1;
  }

  // BOTH OFF → LOST LINE → use memory
  else if (left == 1 && right == 1) {
    if (last_position == -1) {
      turnLeft();   // keep searching left
    }
    else if (last_position == 1) {
      turnRight();  // keep searching right
    }
    else {
      stopMotors(); // no idea → stop (safe)
    }
  }// BOTH ON LINE → go straight
if (left == 1 && right == 1) {
  moveForward();
  last_position = 0;
}

// LEFT SENSOR OFF → drifted right → turn LEFT
else if (left == 0 && right == 1) {
  turnLeft();
  last_position = -1;
}

// RIGHT SENSOR OFF → drifted left → turn RIGHT
else if (left == 1 && right == 0) {
  turnRight();
  last_position = 1;
}

// BOTH OFF → LOST LINE → use memory
else if (left == 0 && right == 0) {
  if (last_position == -1) {
    turnLeft();
  }
  else if (last_position == 1) {
    turnRight();
  }
  else {
    stopMotors();
  }
}
}

void moveForward() {
  analogWrite(a1, speed);
  digitalWrite(a2, LOW);
  digitalWrite(b1A, LOW);
  analogWrite(b2A, speed);
}

void turnLeft() {
  analogWrite(a1, turn_speed);
  digitalWrite(a2, LOW);
  digitalWrite(b1A, LOW);
  analogWrite(b2A, speed);
}

void turnRight() {
  analogWrite(a1, speed);
  digitalWrite(a2, LOW);
  digitalWrite(b1A, LOW);
  analogWrite(b2A, turn_speed);
}

void stopMotors() {
  digitalWrite(a1, LOW);
  digitalWrite(a2, LOW);
  digitalWrite(b1A, LOW);
  digitalWrite(b2A, LOW);
}
