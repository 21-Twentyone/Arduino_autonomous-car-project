Objetivo

Implementar un sistema de navegación autónoma que permita al coche seguir una línea en el suelo mediante el uso de un sensor infrarrojo.

⚙️ Concepto de funcionamiento

El módulo de seguimiento de línea utiliza un sensor IR para detectar diferencias de color basadas en la reflexión:

Superficie oscura (negra) → absorbe luz → detectada como línea
Superficie clara (blanca) → refleja luz → detectada como fondo

El Arduino lee constantemente este valor y ajusta el movimiento del coche en tiempo real.

Lá lógica detras del functionamiento es que el arduino hace que el motor siga una linea y tambien realiza pequeños giros(ajustes) cuando es necesario para volver a la linea.

##nueva material usado: Sensor de linea basado de Infrarojo

##código
const int a1 = 6;
const int a2 = 5;
const int go= 12; //green light
const int b1A = 10;
const int b2A = 9;
const int linetrack=2;

int little_delay=10;
int speed=150; // the velocity at which the wheels spin 
int turn_speed=100;

void moveforward(){
  analogWrite(a1,speed);
  digitalWrite(a2,0);
  digitalWrite(b1A,0);
  analogWrite(b2A,speed);
  digitalWrite(go,HIGH);
}

void turn_left(){
  // Right wheel moves forward faster
  analogWrite(a1,turn_speed);
  analogWrite(a2,0);

  // Left wheel spins slower
  analogWrite(b1A,0);
  analogWrite(b2A,0);

  digitalWrite(go,HIGH);
}

void turn_right(){
  // Left wheel moves forward faster
  analogWrite(b1A,0);
  analogWrite(b2A,turn_speed); 

  // Right wheel spins slower
  analogWrite(a1,0);
  analogWrite(a2,0);

  digitalWrite(go,HIGH);
}

void stop_move(){
  analogWrite(a1,0);
  analogWrite(a2,0);
  analogWrite(b1A,0);
  analogWrite(b2A,0);
  digitalWrite(go,LOW);
}

void setup() {
  Serial.begin(9600);

  pinMode(a1, OUTPUT);
  pinMode(a2, OUTPUT); 
  pinMode(b1A, OUTPUT);
  pinMode(b2A, OUTPUT);
  pinMode(go,OUTPUT);
  pinMode(linetrack,INPUT);
}

void loop() { //the goal is for the car to move in the figure "8" using the linetracking module

int lineColour=digitalRead(linetrack);
Serial.println(lineColour);

if (lineColour==1){ //line detected, so value is equal to 1
  moveforward();

}else{ //no line detected.
  turn_left();
}


}
